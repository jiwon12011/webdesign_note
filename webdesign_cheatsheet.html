🔴 시작하자마자 무조건 먼저!
① <html> lang 변경
<html lang="ko">
② <title> 아래 CSS 링크 삽입
<link rel="stylesheet" href="css/style.css">
③ CSS 링크 바로 아래 jQuery + JS
<script src="제이쿼리1.12.3 경로"></script>
<script src="js/script.js"></script>
⚠️ 순서 중요: CSS link → jQuery → script.js
위치: <head> 안, </head> 닫히기 전!
style.css 맨 위 — 항상 붙여넣기 
@charset "UTF-8";

* {
  margin: 0;
  padding: 0;
  box-sizing: border-box;
}
li     { list-style: none; }
a      { text-decoration: none; color: inherit; }
img    { vertical-align: top; max-width: 100%; }
button { cursor: pointer; }
⚠️ @charset 빠뜨리면 한글 깨짐! 반드시 첫 줄에!
🔴 기본 래퍼 — 반드시 이 안에 작성! 
$(function(){

  // 모든 JS 코드는 여기 안에!

});
이 바깥에 코드 쓰면 동작 안 함!
기능별 코드 조각
탭메뉴 hover 
$("nav > ul > li")
  .mouseover(function(){
    $(".sub")
      .stop().slideDown();
  })
  .mouseout(function(){
    $(".sub")
      .stop().slideUp();
  });
팝업 열기 / 닫기 
$(".pop").click(function(){
  $(".popup")
    .addClass("on");
});

$(".close button")
  .click(function(){
  $(".popup")
    .removeClass("on");
});
슬라이드 — 페이드인아웃 
$('.slide a:gt(0)').hide();  // 첫 이미지 외 숨김

setInterval(function(){
  $(".slide a:first-child")
    .fadeOut()
    .next('a').fadeIn()
    .end()
    .appendTo('.slide');
}, 3000);  // 3초 간격
첫 이미지 빼고 나머지 hide() 후 시작!
슬라이드 — 좌우 이동 
let count = 0;

setInterval(function(){
  count++;
  if(count > 2){ count = 0; }

  $(".slide").css("transform",
    `translateX(${ -33.333 * count }%)`
  );
}, 2500);  // 2.5초 간격
이미지 3장 기준 -33.333% / n장이면 -(100÷n)%
하나씩 내려오는 탭메뉴 
$(".main_menu")
  .mouseover(function(){
    $(this).children(".sub_menu")
      .addClass("show");
  })
  .mouseout(function(){
    $(this).children(".sub_menu")
      .removeClass("show");
  });
각 메뉴마다 독립 동작 → $(this) 사용!
A · B 유형 — 헤더 메뉴 z-index 
/* 슬라이드 이미지에 가려지지 않게! */
nav, header nav ul {
  position: relative;
  z-index: 999;
}
메뉴가 슬라이드에 가려지는 버그 → z-index로 해결!
C · D 유형 — 좌우 flex 분할 
/* 전체화면 좌우 분할 */
.wrap { display: flex; }
.left  { width: 200px; }  /* 메뉴 */
.right { flex: 1; }        /* 콘텐츠 */

/* 서브메뉴 펼치는 방향에 따라 */
/* 오른쪽으로: */ position: absolute;
/* 아래쪽으로: */ position: relative;
전체화면을 정확히 반반 나누는 게 핵심!
F 유형 — 전체 배경 vs 중앙 콘텐츠 
/* 브라우저 전체 배경 */
.full-bg { width: 100%; }

/* 중앙 정렬 콘텐츠 영역 */
.inner {
  width: 1200px;  /* 지정 너비 */
  margin: 0 auto;
}
full-bg(100%) ↔ inner(margin:0 auto) 반드시 구분!
제출 전 최종 체크리스트

항목 누르면 체크!

  • <html lang="ko"> 확인 HTML
  • CSS link 태그 — head 안, title 아래 HTML
  • jQuery → script.js 순서 확인 HTML
  • @charset "UTF-8"; 첫 줄 확인 CSS
  • CSS 초기화 코드 붙여넣었는지 CSS
  • 모든 JS → $(function(){}) 안에 작성 JS
  • 슬라이드 정상 작동 확인 JS
  • 팝업 열기 / 닫기 확인 JS
  • 메뉴 z-index 확인 (A·B 유형) 유형
  • display:flex 좌우 분할 확인 (C·D 유형) 유형
